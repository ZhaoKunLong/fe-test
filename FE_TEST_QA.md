## 算法题
- 请实现一个二分查找算法。
   ```javascript
function find(ary: number[], target: number) {
    if (!ary || ary.length === 0) {
      return false;
    }

    ary.sort((a, b) => a - b);

    let left = 0,
      right = ary.length - 1;

    while (left < right) {
      const mid = Math.floor((left + right) / 2);

      if (ary[mid] === target) {
        return true;
      } else if (ary[left] === target) {
        return true;
      } else if (ary[right] === target) {
        return true;
      } else if (ary[mid] > target) {
        right = mid;
      } else {
        left = mid;
      }
    }

    return false;
  }

   ```
- 请实现一个链表的反转算法。
   ```javascript

class ListNode {
  next: ListNode;
  val: number | string;

  constructor(val?: string | number, next?: ListNode | null) {
    this.next = next ?? null;
    this.val = val ?? 0;
  }
}



 function reverseListNode(head: ListNode | null): ListNode | null {
    if (!head) {
      return null;
    }

    let pre = null;

    while(head) {
      const temp = head.next;
      head.next = pre;
      pre = head;
      head = temp;
    }

    return pre;
  }
   ```
